      SELECT DISTINCT
        M.MODELIDENTIFIER AS "Model ID",
        M.NAME AS "Model Name",    
        LS.STYLENUMBER AS "PCX Dev Style Number",
        B.PCXDEVCOLORWAYID AS "PCX Dev Colorway ID",
        B.MATERIALVERSION as "Material Version",    
        B.PCXBOMID AS "PCX BOM ID",
        DS.ALIAS AS "PCM DPA",
        B.FWPCMBOMID AS "PCM BOM ID",
        CY.ABBREVIATION AS "Season",
        F.FACTORY AS "PCC",
        U.NAME AS "PCC DEVELOPER NTID",
        U.FULLNAME AS "PCC DEVELOPER"    
      FROM
        WCADMIN.FWDEVELOPMENTSTREAM DS,
        WCADMIN.FWBOM B,
        WCADMIN.FWBOMPRODUCTLINK BPL,
        WCADMIN.FWPRODUCT P, 
        WCADMIN.FWLEGACYSTYLE LS,
        WCADMIN.FWDEVSTREAMFACTORYRESOURCE DSF,
        WCADMIN.FWFACTORY FWF,
        WCADMIN.FACTORY F,
        WCADMIN.FWCYCLEYEAR CY,
        WCADMIN.FWMODEL M,
        WCADMIN.ROLEPRINCIPALMAP RPM,
        WCADMIN.TEAM T,
        WCADMIN.WTUSER U
      WHERE
        DS.IDA2A2 = B.IDA3B9
        AND B.IDA2A2 = BPL.IDA3A5
        AND BPL.IDA3B5 = P.IDA2A2
        AND P.IDA3K9 = LS.IDA2A2
        AND P.IDA3L9 = CY.IDA2A2
        AND DS.IDA2A2 = DSF.IDA3A5
        AND DSF.IDA3B5 = FWF.IDA2A2
        AND FWF.IDA3A5 = F.IDA2A2
        AND P.IDA3B9 = M.IDA2A2    
        AND RPM.IDA3A4 = T.IDA2A2 
        AND T.IDA2A2 = DS.IDA3TEAMID
        AND RPM.IDA3B4 = U.IDA2A2(+) 
        AND RPM.ROLE = 'FWPCCDEVELOPER'
        AND DSF.FACTORYTYPE = 'DEV'
        AND CY.ABBREVIATION IN ('SU20', 'FA20', 'HO20', 'SP21')
        AND F.FACTORY IN ('T2','VT');
      

      
      SELECT DISTINCT
        devstyle.ptc_str_15typeinfolcsproduct            AS "Model ID",
        devstyle.ptc_str_17typeinfolcsproduct            AS "Model Name",
        devstyle.ptc_str_32typeinfolcsproduct            AS "Dev Style number",
        devstyle.ptc_str_20typeinfolcsproduct            AS "Marketing Name",
        colorway.ptc_lng_1typeinfolcssku                 AS "Colorway ID",
        colorway.ptc_str_1typeinfolcssku                 AS "Colorway Name",
        upper(skuseasonlink.ptc_str_13typeinfolcsskuseas) AS "Product C/O Status",
        upper(colorway.ptc_str_2typeinfolcssku) AS "Dev Colorway State",
        upper(colorway.ptc_str_19typeinfolcssku) AS "Dev Colorway Gate",
        season.ptc_str_4typeinfolcsseason                AS "Season",
        upper(devstyle.ptc_str_10typeinfolcsproduct) AS "Gender",
        upper(ptc_str_2typeinfolcsproduct) AS "Age",
        upper(devstyle.ptc_str_12typeinfolcsproduct) AS "Sports Activity",
        productseasonlink.ptc_str_14typeinfolcsproduct   AS "Development Team",
        (
            SELECT
                lcslifecyclemanaged.ptc_str_10typeinfolcslifecyc
            FROM
                manrev.lcslifecyclemanaged lcslifecyclemanaged
            WHERE
                productseasonlink.ida3a2typeinfolcsproductseas = lcslifecyclemanaged.ida2a2
        ) AS "TD Code",
        supplier.ptc_str_24typeinfolcssupplie            AS "PCC",
        supplier.ptc_str_60typeinfolcssupplie            AS "Factory Group",
        sourcingconfig.ptc_str_2typeinfolcssourcing      AS "Primary Source"
      FROM
        manrev.lcsproduct devstyle,
        manrev.lcsproductseasonlink productseasonlink,
        manrev.lcssku colorway,
        manrev.lcsseason season,
        manrev.lcsskuseasonlink skuseasonlink,
        manrev.lcssourcingconfig sourcingconfig,
        manrev.lcssourcetoseasonlink sourcetoseasonlink,
        manrev.lcssupplier supplier
      WHERE
        season.ptc_str_4typeinfolcsseason IN (
            'Footwear-SP20',
            'Footwear-SU20',
            'Footwear-FA20',
            'Footwear-HO20',
            'Footwear-SP21',
            'Footwear-SU21',
            'Footwear-FA21',
            'Footwear-HO21',
            'Footwear-SP22'
        )
        AND sourcetoseasonlink.ptc_bln_1typeinfolcssourceto = 1 --Only Primary Source (Season)
        AND upper(colorway.ptc_str_13typeinfolcssku) = 'ACTIVE'--Only active Colorways
        AND devstyle.ida3masterreference = colorway.ida3a12
        AND devstyle.flextypeidpath = '\49742\250384\250572\69159615' -- FW Dev Styles
        --AND devstyle.flextypeidpath = '\49742\250384\250572\69159185' -- Construction Option Dev Styles   
        --AND devstyle.flextypeidpath = '\49742\250384\250572\69159400' -- Promo Dev Styles
        --AND skuseasonlink.ptc_str_13typeInfoLCSSKUSeas = 'new'
        AND devstyle.ida3masterreference = productseasonlink.productmasterid
        AND devstyle.ida3masterreference = skuseasonlink.productmasterid
        AND colorway.branchiditerationinfo = skuseasonlink.skuarevid
        AND colorway.ida3masterreference = skuseasonlink.skumasterid
        AND productseasonlink.seasonrevid = season.branchiditerationinfo
        AND productseasonlink.ida3b5 = season.ida3masterreference
        AND productseasonlink.ida3b5 = skuseasonlink.ida3b5
        AND productseasonlink.productmasterid = skuseasonlink.productmasterid
        AND skuseasonlink.ida3b5 = season.ida3masterreference
        AND skuseasonlink.seasonrevid = season.branchiditerationinfo
        AND devstyle.ida3masterreference = sourcingconfig.ida3c11
        AND devstyle.branchiditerationinfo = sourcingconfig.productarevid
        AND sourcetoseasonlink.seasonmasterid = season.ida3masterreference
        AND sourcetoseasonlink.seasonmasterid = skuseasonlink.ida3b5
        AND supplier.branchiditerationinfo = sourcetoseasonlink.branchida3b2typeinfolcssourc
        AND sourcetoseasonlink.sourcingconfigmasterid = sourcingconfig.ida3masterreference
        AND productseasonlink.ida3a2typeinfolcsproductseas IN (
            14342608,--F1
            14342631,--F2
            14342651,--F3
            14342674--F4
        )--TD Code (add if required N1-14342353, N2-14342369, N3-14342389)
        /*AND supplier.ptc_str_60typeinfolcssupplie IN (
            'POU CHEN',
            'FENG TAY',
            'TAE KWANG',
            'CHANG SHIN'
        )*/
        AND colorway.ptc_lng_1typeinfolcssku <> 0
        AND devstyle.latestiterationinfo = 1
        AND devstyle.versionida2versioninfo = 'A'
        AND colorway.latestiterationinfo = 1
        AND colorway.versionida2versioninfo = 'A'
        AND season.latestiterationinfo = 1
        AND season.versionida2versioninfo = 'A'
        AND productseasonlink.seasonrevid = season.branchiditerationinfo
        AND devstyle.ida3masterreference = sourcingconfig.ida3c11
        AND devstyle.branchiditerationinfo = sourcingconfig.productarevid
        AND sourcingconfig.latestiterationinfo = 1
        AND sourcingconfig.versionida2versioninfo = 'A'
        AND productseasonlink.effectlatest = 1
        AND skuseasonlink.effectlatest = 1
        AND sourcetoseasonlink.latestiterationinfo = 1
        AND sourcetoseasonlink.versionida2versioninfo = 'A'
        AND supplier.latestiterationinfo = 1
        AND supplier.versionida2versioninfo = 'A'
      ORDER BY
        season.ptc_str_4typeinfolcsseason,
        devstyle.ptc_str_15typeinfolcsproduct,
        devstyle.ptc_str_32typeinfolcsproduct,
        colorway.ptc_lng_1typeinfolcssku;

        TAB.HIER_PATH
        ,REPLACE(TAB.HIER_DISPLAY_PATH,'\\','') AS HIERARCHY_DISPLAY_PATH
        ,REGEXP_REPLACE(TAB.HIER_PATH_NAME,'\\',NULL,1,1) AS HIER_PATH_NAME

        FROM (
        SELECT (a.IDA3MASTERREFERENCE+1) as IDA3MASTERREFERENCE
        ,INSTANTIABLE 
        ,LEVEL
        ,SYS_CONNECT_BY_PATH(a.IDA3MASTERREFERENCE+1, '\') AS HIER_PATH

        ,SYS_CONNECT_BY_PATH(a.HIER_DISP_NAME1, '\') AS HIER_DISPLAY_PATH 
        ,a.DISPLAYNAMEKEY 
        ,a.HIERARCHYDISPLAYNAMEKEY
        FROM (SELECT wtd.IDA2A2
        ,wtd.IDA3PARENTREFERENCE
        ,wtd.IDA3MASTERREFERENCE
        ,wtd.INSTANTIABLE
        ,wtdm.DISPLAYNAMEKEY 
        ,wtdm.HIERARCHYDISPLAYNAMEKEY
        ,wtdm.HIER_DISP_NAME1
        FROM manrev.WTTYPEDEFINITION wtd 
        INNER JOIN ( 
        SELECT 
        wtdm1.IDA2A2
        ,wtdm1.DISPLAYNAMEKEY 
        ,wtdm1.HIERARCHYDISPLAYNAMEKEY
        ,HDNAME.HIER_DISP_NAME1
        FROM manrev.WTTYPEDEFINITIONMASTER wtdm1 
        LEFT OUTER JOIN (select TYPE_DEF.name as HIERARCHYDISPLAYNAMEKEY
        ,LOC_PROP_VAL.VALUE AS HIER_DISP_NAME1
        from manrev.LWCTYPEDEFINITION TYPE_DEF
        INNER JOIN manrev.LWCLOCALIZABLEPROPERTYVALUE LOC_PROP_VAL ON TYPE_DEF.IDA2A2 = LOC_PROP_VAL.IDA3C4 
        and LOC_PROP_VAL.CLASSNAMEKEYC4 = 'com.ptc.core.lwc.server.LWCTypeDefinition'
        and LOC_PROP_VAL.CLASSNAMEKEYA4 = 'com.ptc.core.lwc.server.LWCPropertyDefinition' 
        and LOC_PROP_VAL.IDA3A4 = 2292) HDNAME ON HDNAME.HIERARCHYDISPLAYNAMEKEY = wtdm1.HIERARCHYDISPLAYNAMEKEY
        ) wtdm 
        on wtdm.IDA2A2 = wtd.IDA3MASTERREFERENCE
        WHERE wtd.LATESTITERATIONINFO = 1
        ) a
        WHERE level <= 20 
        START WITH a.IDA3PARENTREFERENCE = 0
        CONNECT BY NOCYCLE PRIOR a.IDA2A2 = a.IDA3PARENTREFERENCE AND LEVEL <= 20) TAB
        WHERE REPLACE(TAB.HIER_DISPLAY_PATH,'\\','') LIKE 'Nike\Apparel%'
        )
        SELECT DISTINCT 
        HIER_DEF_PATH.SM_MAPPINGTAB AS MAPPINGTAB
        ,LCSMATERIAL.PTC_LNG_1TYPEINFOLCSMATERIAL AS PDM_MATERIAL_NUMBER /* NikeLegacyCode NEED_COLUMN_UPDATE_CHECK */

        ,HIER_DEF_PATH.PDM_LONGDESC
        ,HIER_DEF_PATH.HIER_PATH_NAME
        FROM MANREV.LCSMATERIALSUPPLIER@PTCFLEX
        INNER JOIN MANREV.LCSMATERIALSUPPLIERMASTER@PTCFLEX ON LCSMATERIALSUPPLIER.IDA3MASTERREFERENCE = LCSMATERIALSUPPLIERMASTER.IDA2A2
        INNER JOIN MANREV.LCSMATERIAL@PTCFLEX ON LCSMATERIAL.IDA3MASTERREFERENCE = LCSMATERIALSUPPLIERMASTER.IDA3A6
        INNER JOIN MANREV.LCSMATERIALMASTER@PTCFLEX ON LCSMATERIALMASTER.IDA2A2 = LCSMATERIAL.IDA3MASTERREFERENCE
        INNER JOIN MANREV.LCSSUPPLIERMASTER@PTCFLEX ON LCSSUPPLIERMASTER.IDA2A2 = LCSMATERIALSUPPLIERMASTER.IDA3B6
        INNER JOIN MANREV.LCSSUPPLIER@PTCFLEX ON LCSSUPPLIERMASTER.IDA2A2 = LCSSUPPLIER.IDA3MASTERREFERENCE

        AND LCSSUPPLIERMASTER.PLACEHOLDER = 0
